1. Why this stack?
Requirements you mentioned

    Runs locally on a machine.

    Single page with input fields for simulation parameters.

    Shows some fixed images.

    Does sanity checks on the inputs.

    Generates a JSON file (likely downloaded or shown to the user).

All of that can be done completely in the browser with:

    HTML → structure (forms, input boxes, layout, images).

    CSS (or small framework like Tailwind/Bootstrap) → layout, styling.

    JavaScript → validation, JSON creation, trigger file download.

No need for React, Angular, Vue, etc. unless:

    You expect this to grow into a complex web app.

    Or you want reusable components, routing, large-scale state management.

Given your HPC / GPU / C++ background, keeping this minimal will be much faster to implement and easier to maintain.
2. Recommended stack in practice
Option A (simplest, zero tooling)

    HTML5: index.html

    Vanilla JS: <script> tag in the same file or main.js.

    CSS:

        Either a simple style.css, or

        Include a CDN for Tailwind CSS or Bootstrap if you want nice styling without effort.

How to run:

    Just open index.html with your browser (double-click).

    For some browsers, saving files via Blob works fine even from file://.

    If you run into any restrictions, you can launch a tiny local server:

        python -m http.server 8000 in the folder, then open http://localhost:8000/index.html.

Option B (slightly more structured, still lightweight)

If you want TypeScript + better dev experience:

    Vite + TypeScript + Vanilla (no framework)

        npm create vite@latest my-app -- --template vanilla-ts

        Implement your logic in main.ts.

        npm run build → produces static files in dist/ you can copy anywhere.

This adds a build step, but you still ship plain HTML+JS in the end.

4. Minimal working skeleton (pure HTML + JS)

Here’s a compact, correct example you can adapt.
Save as index.html and open in a browser:

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <title>Simulation Parameters</title>
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <!-- Optional: quick nice styling via CDN (Bootstrap example) -->
  <link
    rel="stylesheet"
    href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css"
  />
</head>
<body class="bg-light">
  <div class="container py-4">
    <h1 class="mb-4">Simulation Parameters</h1>

    <!-- Fixed images row -->
    <div class="mb-4 d-flex gap-3">
      <img src="image1.png" alt="Simulation diagram 1" style="max-height: 150px;" />
      <img src="image2.png" alt="Simulation diagram 2" style="max-height: 150px;" />
    </div>

    <!-- Form -->
    <form id="sim-form" class="card p-3">
      <div class="row g-3">
        <div class="col-md-4">
          <label for="paramName" class="form-label">Simulation name</label>
          <input type="text" id="paramName" class="form-control" required />
        </div>

        <div class="col-md-4">
          <label for="timeStep" class="form-label">Time step (s)</label>
          <input type="number" id="timeStep" class="form-control" step="0.0001" required />
        </div>

        <div class="col-md-4">
          <label for="numParticles" class="form-label">Number of particles</label>
          <input type="number" id="numParticles" class="form-control" min="1" required />
        </div>

        <div class="col-md-4">
          <label for="material" class="form-label">Material</label>
          <select id="material" class="form-select">
            <option value="AA7075">AA7075</option>
            <option value="AA6061">AA6061</option>
            <option value="AA5083">AA5083</option>
          </select>
        </div>
      </div>

      <div id="errorBox" class="mt-3 text-danger"></div>

      <div class="mt-4">
        <button type="button" id="generateBtn" class="btn btn-primary">
          Generate JSON
        </button>
      </div>
    </form>
  </div>

  <script>
    function sanityCheck(params) {
      const errors = [];

      if (!params.name || params.name.trim().length === 0) {
        errors.push("Simulation name is required.");
      }

      if (isNaN(params.timeStep) || params.timeStep <= 0) {
        errors.push("Time step must be a positive number.");
      } else if (params.timeStep > 1) {
        errors.push("Time step seems too large (> 1 s).");
      }

      if (isNaN(params.numParticles) || params.numParticles <= 0) {
        errors.push("Number of particles must be a positive integer.");
      } else if (params.numParticles > 1e7) {
        errors.push("Number of particles seems too large (> 1e7).");
      }

      return errors;
    }

    function downloadJsonFile(obj, fileName) {
      const jsonStr = JSON.stringify(obj, null, 2);
      const blob = new Blob([jsonStr], { type: "application/json" });
      const url = URL.createObjectURL(blob);

      const a = document.createElement("a");
      a.href = url;
      a.download = fileName;
      a.click();

      URL.revokeObjectURL(url);
    }

    document.getElementById("generateBtn").addEventListener("click", () => {
      const params = {
        name: document.getElementById("paramName").value,
        timeStep: parseFloat(document.getElementById("timeStep").value),
        numParticles: parseInt(document.getElementById("numParticles").value, 10),
        material: document.getElementById("material").value,
      };

      const errorBox = document.getElementById("errorBox");
      const errors = sanityCheck(params);

      if (errors.length > 0) {
        errorBox.textContent = errors.join(" ");
        return;
      }

      errorBox.textContent = "";

      // Add metadata if you want
      const output = {
        simulation: params,
        generatedAt: new Date().toISOString(),
        version: "1.0",
      };

      downloadJsonFile(output, "simulation_params.json");
    });
  </script>
</body>
</html>

You can adapt:
