<script>
    const buttons = document.querySelectorAll('.tab-button');
    const contents = document.querySelectorAll('.tab-content');

    buttons.forEach(button => {
        button.addEventListener('click', () => {
            // Alle Inhalte ausblenden
            contents.forEach(c => c.style.display = 'none');
            // Alle Buttons inaktiv machen
            buttons.forEach(b => b.classList.remove('active'));

            // Gewählten Tab anzeigen
            const tab = button.getAttribute('data-tab');
            document.getElementById(tab).style.display = 'block';

            // Button aktiv setzen
            button.classList.add('active');
        });
    });


    document.getElementById("saveJson").addEventListener("click", () => {
        // Alle Tabs und deren Inputs
        const tabs = {
            units: ["length_unit", "time_unit", "mass_unit", "temperature_unit", "angle_unit"],
            sub_rod_dimensions: ["sub_width", "sub_length", "sub_thickness", "rod_height", "rod_diameter"],
            Velocitys: ["vx_sub", "vy_sub", "vz_sub", "rod_feed", "rpm"],
            "constants-substrate": ["E_sub", "nu_sub", "rho_sub", "jc_A_sub", "jc_B_sub", "jc_C_sub", "jc_m_sub", "jc_n_sub", "jc_Tref_sub", "jc_Tmelt_sub", "jc_epsdot_sub", "jc_clamp_sub", "c_sub", "tq_sub", "therm_eff_sub", "k_sub"],
            "constants-rod": ["E_rod", "nu_rod", "rho_rod", "jc_A_rod", "jc_B_rod", "jc_C_rod", "jc_m_rod", "jc_n_rod", "jc_Tref_rod", "jc_Tmelt_rod", "jc_epsdot_rod", "jc_clamp_rod", "c_rod", "tq_rod", "therm_eff_rod", "k_rod"],
            global_constants: ["init_particle_dist", "mass_scale", "smooth_len", "init_temp", "vel_scale", "proc_time"],
            corr_parameters: ["alpha_sph", "beta_sph", "visc_reg", "xsph_corr", "stress_reg"],
            heat_convection: ["h_air", "h_metal"],
            centering_substrate: ["shift_x", "shift_y", "cfl"]
        };

        // JSON Objekt vorbereiten
        let jsonData = {};

        for (let tab in tabs) {
            jsonData[tab] = {};
            tabs[tab].forEach(id => {
                const element = document.getElementById(id);
                if (element) {
                    // Wert aus Input oder Select auslesen
                    jsonData[tab][id] = element.value;
                }
            });
        }

        // JSON in Datei konvertieren und herunterladen
        const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(jsonData, null, 4));
        const dlAnchor = document.createElement('a');
        dlAnchor.setAttribute("href", dataStr);
        dlAnchor.setAttribute("download", "FS_values.json");
        document.body.appendChild(dlAnchor);
        dlAnchor.click();
        dlAnchor.remove();
    });

    document.getElementById("saveJson").addEventListener("click", () => {
        // ... dein bestehender JSON Code ...

        // Erfolgsmeldung anzeigen
        const output = document.getElementById("output-content");
        output.textContent = "Job successfull created!";
    });

</script>
